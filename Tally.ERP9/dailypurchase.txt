[collection :colldailypurchase]
	type: voucher
	filter:fcolldailypurchase
	fetch :*.*,inventoryentries.*,partyledgername
	filter : sgrp
	filter :sled
	[system:formula]
		fcolldailypurchase:$$ispurchase:$vouchertypename and $vouchertypename contains "Milk"
		sgrp: if $$issysname:#groupname then yes else $parent:ledger:$partyledgername=#groupname
		sled: if $$issysname:#ledgername then yes else $partyledgername=#ledgername
